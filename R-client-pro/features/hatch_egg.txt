-- ====================================================================
-- MODULE: AUTO ẤP TRỨNG & THU HOẠCH (DÙNG REMOTE + TRÍ NHỚ NHÂN TẠO)
-- ====================================================================
return function(Window, Utils)
    local Workspace = game:GetService("Workspace")
    local ReplicatedStorage = game:GetService("ReplicatedStorage")
    
    local autoHatchEnabled = false
    local targetEggId = 60      -- ID trứng mặc định
    local hatchTimeMinutes = 60 -- Thời gian ấp mặc định (60 phút)
    
    -- TRÍ NHỚ NHÂN TẠO: Lưu mốc thời gian hoàn thành (Tính bằng Giây os.time())
    local hatchMemory = {} 
    
    -- ==========================================
    -- GIAO DIỆN UI (Tạo Tab mới)
    -- ==========================================
    local HatchTab = Window:CreateTab("Ấp Trứng", 4483362458)
    
    HatchTab:CreateSection("--- THIẾT LẬP AUTO HATCH ---")
    
    HatchTab:CreateInput({
        Name = "Nhập ID Trứng cần ấp",
        PlaceholderText = "Mặc định đang là 60...",
        RemoveTextAfterFocusLost = false,
        Flag = "TargetEggIdInput",
        Callback = function(Text)
            local id = tonumber(Text)
            if id then 
                targetEggId = id 
                print("🥚 Đã cài đặt ID trứng sẽ ấp là:", targetEggId)
            end
        end,
    })

    HatchTab:CreateInput({
        Name = "Thời gian ấp 1 mẻ (Phút)",
        PlaceholderText = "Nhập 60 hoặc 50...",
        RemoveTextAfterFocusLost = false,
        Flag = "TargetEggTimeInput",
        Callback = function(Text)
            local mins = tonumber(Text)
            if mins then 
                hatchTimeMinutes = mins 
                print("⏳ Đã cài đặt thời gian ấp là:", hatchTimeMinutes, "phút")
            end
        end,
    })

    HatchTab:CreateToggle({
        Name = "Bật/Tắt Auto Ấp & Nhận (Cả 5 Máy)",
        CurrentValue = false,
        Flag = "AutoHatchToggle",
        Callback = function(Value)
            autoHatchEnabled = Value
        end,
    })

    -- Đường dẫn gọi Remote của Game
    local DataPullFunc = ReplicatedStorage:WaitForChild("CommonLibrary")
        :WaitForChild("Tool")
        :WaitForChild("RemoteManager")
        :WaitForChild("Funcs")
        :WaitForChild("DataPullFunc")

    -- Hàm phụ trợ: Dịch từ "00:47:29" sang tổng số giây
    local function parseTimeToSeconds(timeStr)
        local parts = string.split(timeStr, ":")
        if #parts == 3 then
            return tonumber(parts[1])*3600 + tonumber(parts[2])*60 + tonumber(parts[3])
        elseif #parts == 2 then
            return tonumber(parts[1])*60 + tonumber(parts[2])
        end
        return 0
    end

    -- ==========================================
    -- VÒNG LẶP CORE: QUẢN LÝ 5 MÁY ẤP
    -- ==========================================
    task.spawn(function()
        while task.wait(1) do
            if not autoHatchEnabled then continue end
            
            local eggHatchFolder = Workspace:FindFirstChild("Area") 
                and Workspace.Area:FindFirstChild("center") 
                and Workspace.Area.center:FindFirstChild("Area") 
                and Workspace.Area.center.Area:FindFirstChild("EggHatch")
                
            if not eggHatchFolder then continue end
            
            local currentTime = os.time()
            
            for i = 1, 5 do
                local machineName = "Hatch_" .. i
                local machine = eggHatchFolder:FindFirstChild(machineName)
                
                if machine then
                    -- NẾU TOOL CHƯA CÓ TRÍ NHỚ VỀ MÁY NÀY (Vừa bật tool hoặc vừa Rejoin)
                    if not hatchMemory[i] then
                        local uiState = "Empty"
                        local uiSecondsLeft = 0
                        
                        -- Lục lọi UI để xem máy đang làm gì
                        local topGui = machine:FindFirstChild("TopGUI")
                        local hatchInfoGui = topGui and topGui:FindFirstChild("HatchInfoGui")
                        
                        if hatchInfoGui then
                            for _, child in ipairs(hatchInfoGui:GetDescendants()) do
                                if child:IsA("TextLabel") and child.Visible and child.Text ~= "" then
                                    local text = string.lower(child.Text)
                                    if string.match(text, "ready") then
                                        uiState = "Ready"
                                    elseif string.match(text, "%d+:%d+") then
                                        uiState = "Hatching"
                                        uiSecondsLeft = parseTimeToSeconds(child.Text)
                                    end
                                end
                            end
                        end
                        
                        -- Đồng bộ vào trí nhớ
                        if uiState == "Hatching" and uiSecondsLeft > 0 then
                            hatchMemory[i] = currentTime + uiSecondsLeft
                            print(string.format("🧠 [TRÍ NHỚ] Đã đồng bộ Máy %d: Đang ấp dở (Còn %d giây)", i, uiSecondsLeft))
                        
                        elseif uiState == "Ready" then
                            -- Nếu trứng đã chín sẵn
                            pcall(function() DataPullFunc:InvokeServer("EggHatchTakenChannel", i) end)
                            task.wait(0.2)
                            pcall(function() DataPullFunc:InvokeServer("EggHatchStartChannel", i, targetEggId) end)
                            
                            hatchMemory[i] = currentTime + (hatchTimeMinutes * 60)
                            print(string.format("🥚 [AUTO] Đã Thu Hoạch & Ấp Mẻ Mới cho Máy %d", i))
                            task.wait(0.5)
                        
                        elseif uiState == "Empty" then
                            -- Nếu máy trống không
                            pcall(function() DataPullFunc:InvokeServer("EggHatchStartChannel", i, targetEggId) end)
                            hatchMemory[i] = currentTime + (hatchTimeMinutes * 60)
                            print(string.format("🥚 [AUTO] Bắt Đầu Ấp máy %d (ID: %d - Đợi %d phút)", i, targetEggId, hatchTimeMinutes))
                            task.wait(0.5)
                        end
                        
                    -- NẾU TOOL ĐÃ CÓ TRÍ NHỚ (Chỉ tin vào trí nhớ, PHỚT LỜ UI!)
                    else
                        if currentTime >= hatchMemory[i] then
                            -- Đúng hẹn đồng hồ báo thức -> Gửi Combo lệnh (Thu Hoạch -> Ấp Tiếp)
                            pcall(function() DataPullFunc:InvokeServer("EggHatchTakenChannel", i) end)
                            task.wait(0.2)
                            pcall(function() DataPullFunc:InvokeServer("EggHatchStartChannel", i, targetEggId) end)
                            
                            hatchMemory[i] = currentTime + (hatchTimeMinutes * 60)
                            print(string.format("🌟 [AUTO] HOÀN THÀNH ẤP Máy %d! Đã thu hoạch và Tự động ấp mẻ mới.", i))
                            task.wait(0.5)
                        end
                    end
                end
            end
        end
    end)
end
