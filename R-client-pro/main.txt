-- ====================================================================
-- MAIN SCRIPT: Khởi chạy R-Client Pro
-- ====================================================================
local Rayfield = loadstring(game:HttpGet('https://sirius.menu/rayfield'))()

local Window = Rayfield:CreateWindow({
    Name = "R-Client Pro Tool v2.5",
    LoadingTitle = "Đang tải hệ thống Core...",
    LoadingSubtitle = "Dev by Sơn",
    ConfigurationSaving = { Enabled = true, FileName = "AutoFarmConfig" },
    KeySystem = false
    DisableRayfieldPrompts = true,
    DisableBuildWarnings = true 
})

-- Kéo các file từ GitHub về và chạy ngầm (Đã sửa lại đường dẫn chuẩn xác)
local Utils = loadstring(game:HttpGet('https://raw.githubusercontent.com/chaocauminhlason/scripts-linh-tinh/main/R-client-pro/core/utilities.txt'))()
local WebhookModule = loadstring(game:HttpGet('https://raw.githubusercontent.com/chaocauminhlason/scripts-linh-tinh/main/R-client-pro/core/webhook.txt'))()

-- Khởi tạo tab ghim cấu hình Webhook
if WebhookModule then
    WebhookModule.InitTab(Window, Rayfield)
end

-- Tải các phân mảnh tính năng vào giao diện UI chính
if Utils then
    loadstring(game:HttpGet('https://raw.githubusercontent.com/chaocauminhlason/scripts-linh-tinh/main/R-client-pro/features/tracker.txt'))()(Window, Utils)
    loadstring(game:HttpGet('https://raw.githubusercontent.com/chaocauminhlason/scripts-linh-tinh/main/R-client-pro/features/rgame_farm.txt'))()(Window, Utils)
    loadstring(game:HttpGet('https://raw.githubusercontent.com/chaocauminhlason/scripts-linh-tinh/main/R-client-pro/features/boss_hunt.txt'))()(Window, Utils)
    
    -- Đã sửa từ dungeon_auto.txt thành auto_dungeon.txt đúng chuẩn tên file của bạn
    if WebhookModule then
        loadstring(game:HttpGet('https://raw.githubusercontent.com/chaocauminhlason/scripts-linh-tinh/main/R-client-pro/features/auto_dungeon.txt'))()(Window, Utils, WebhookModule)
    end
    
    loadstring(game:HttpGet('https://raw.githubusercontent.com/chaocauminhlason/scripts-linh-tinh/main/R-client-pro/features/optimization.txt'))()(Window, Utils)
end

Rayfield:LoadConfiguration()
Rayfield:Notify({ Title = "Thành Công!", Content = "Hệ thống R-Client Pro đã nạp đầy đủ Modules!", Duration = 5 })
